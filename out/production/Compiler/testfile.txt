int testForExpr(int num1, int num2) {
	if (num1 >= num2) {
		return 0;
	}
	int j, i;
	int ans1 = 1, ans2 = 1;
	for (i = 1, j = 0; i < num1; i = i + 1, j = j + 1) {
		ans1 = ans1 * i;
	}
	for (i = 0; j < num2; ) {
		ans2 = ans2 * j + i;
		i = i + 1;
		j = j + 1;
	}
	printf("%d %d\n", i, j);
	return ans1 + ans2;
}
