int factorial(int n) {
    if (n < 0) {
        printf("error: n = %d < 0\n", n);
        return 0;
    }
    if (n == 0) {
        return 1;
    }
    return n * factorial(n - 1);
}

void swap0(int a[], int b[]) {
    int t = a[0];
    a[0] = b[0];
    b[0] = t;
}

int copyEven(int dst[], int src[], int len) {
    int ret = 0;
    int i = 0;
    for (i = 0; i < len; i = i + 1) {
        if (src[i] % 2 == 0) {
            dst[i] = src[i];
            ret = ret + 1;
        }
    }
    return ret;
}

int map(int num) {
    static int map_cnt;
    map_cnt = map_cnt + 1;
    return (31 + num + map_cnt) * 5 % 17;
}

void getArr(int dst[], int n) {
    int i;
    for (i = 0; i < n; i = i + 1) {
