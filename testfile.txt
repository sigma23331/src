void func(int a) {
    // 此时 'a' 在 $a0 中

    printf("999\n");
    // buildPutintInst 生成: li $a0, 999; syscall
    // 此时 $a0 变成了 999！参数 'a' 的值丢失了！

    int b = a + 1;
    printf("%d",b);
    // 编译器查看 var2reg，发现 'a' 映射在 $a0。
    // 于是生成: move $reg, $a0 (值为 999)
    // 结果 b = 1000，而不是 a + 1。
}

int main() {
    int a = 1;
    func(a);
    return 0;
}
